# Suscribirse a un topic con mosquitto
    docker exec -it mosquitto mosquitto_sub -h localhost -p 1883 -t "testmq" -v

# Publicar a un topic con mosquitto
    docker exec -it mosquitto mosquitto_pub -h localhost -p 1883 -t "testmq" -m "holamundo"

# Conectar el canal "mqtt" mediante chasgimqtt con el broker "mosquitto" y que
# transmita todo lo relacionado con el tópico "testmq"
    docker exec -it totems-web chasgimqtt -H mosquitto -p 1883 --topic=testmq:2 -n mqtt site_project.asgi:channel_layer -v

# Ejecutar los workers
    docker exec -it totems-web python3 manage.py runworker mqtt
